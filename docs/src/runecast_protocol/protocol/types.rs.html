189/// State of the timer vote system.
190///
191/// The timer vote allows players to collectively vote to start a turn timer
192/// on the current player. This prevents indefinite stalling.
193#[derive(Debug, Clone, Serialize, Deserialize, Default)]
194#[serde(tag = "status", rename_all = "snake_case")]
195pub enum TimerVoteState {
196    /// No vote in progress, button is idle
197    #[default]
198    Idle,
199
200    /// Vote is in progress
201    VoteInProgress {
202        /// User ID of who initiated the vote
203        initiator_id: String,
204        /// User IDs of players who have voted yes
205        voters: Vec<String>,
206        /// Total votes needed to pass
207        votes_needed: u32,
208        /// When the vote expires
209        expires_at: chrono::DateTime<chrono::Utc>,
210    },
211
212    /// Timer is actively counting down
213    TimerActive {
214        /// When the timer expires
215        expires_at: chrono::DateTime<chrono::Utc>,
216        /// Target player ID (user ID)
217        target_player_id: String,
218    },
219
220    /// Vote failed, in cooldown before another can start
221    Cooldown {
222        /// When the cooldown expires
223        expires_at: chrono::DateTime<chrono::Utc>,
224    },
225
226    /// Feature disabled (not enough players)
227    Disabled,
228}

246    },
247
248    /// A player's ready state changed
249    PlayerReadyChanged { player_id: String, is_ready: bool },
250
251    /// A player's connection state changed
252    PlayerConnectionChanged {
253        player_id: String,
254        is_connected: bool,
255    },
256
257    /// A game's state changed
258    GameStateChanged { game_id: String, state: GameState },
259
260    /// Queue count updated for a game
261    QueueUpdated { game_id: String, queue_count: u32 },
262
263    /// Host changed
264    HostChanged { new_host_id: String },
265}

282        score: i32,

284    },
285
286    /// Turn changed to another player
287    TurnChanged { player_id: String },
288
289    /// Round number changed

291
292    /// A word was added to used words
293    WordUsed { word: String },
294
295    /// A spectator joined the game

297
298    /// A spectator left the game

301    /// A player's connection state changed

325/// Standard error codes for protocol errors.
326#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
327#[serde(rename_all = "snake_case")]
328pub enum ErrorCode {
329    // Connection errors
330    NotAuthenticated,
331    SessionExpired,
332    InvalidSession,
333
334    // Lobby errors
335    LobbyNotFound,
336    LobbyFull,
337    NotInLobby,
338    AlreadyInLobby,
339
340    // Game errors
341    GameNotFound,
342    GameInProgress,
343    GameNotActive,
344    NotInGame,
345    AlreadyInGame,
346
347    // Turn errors
348    NotYourTurn,
349    InvalidAction,
350    ActionTimeout,
351
352    // Word submission errors
353    InvalidPath,
354    PathTooShort,
355    WordNotInDictionary,
356    WordAlreadyUsed,
357
358    // Permission errors
359    NotHost,
360    NotEnoughPlayers,
361    TooManyPlayers,
362
363    // Resource errors
364    InsufficientGems,
365
366    // Rate limiting
367    TooManyRequests,
368    MessageTooLarge,
369
370    // Generic
371    InvalidRequest,
372    InternalError,
373}
374
375impl ErrorCode {
376    /// Get a human-readable message for this error code.
377    pub fn message(&self) -> &'static str {
378        match self {
379            Self::NotAuthenticated => "Not authenticated",
380            Self::SessionExpired => "Session expired",
381            Self::InvalidSession => "Invalid session",
382            Self::LobbyNotFound => "Lobby not found",
383            Self::LobbyFull => "Lobby is full",
384            Self::NotInLobby => "You must be in a lobby",
385            Self::AlreadyInLobby => "Already in a lobby",
386            Self::GameNotFound => "Game not found",
387            Self::GameInProgress => "A game is already in progress",
388            Self::GameNotActive => "Game is not active",
389            Self::NotInGame => "You are not in this game",
390            Self::AlreadyInGame => "You are already in this game",
391            Self::NotYourTurn => "It's not your turn",
392            Self::InvalidAction => "Invalid action",
393            Self::ActionTimeout => "Action timed out",
394            Self::InvalidPath => "Invalid path - letters must be adjacent",
395            Self::PathTooShort => "Word must be at least 3 letters",
396            Self::WordNotInDictionary => "Word not found in dictionary",
397            Self::WordAlreadyUsed => "Word has already been used",
398            Self::NotHost => "Only the host can do this",
399            Self::NotEnoughPlayers => "Not enough players",
400            Self::TooManyPlayers => "Too many players",
401            Self::InsufficientGems => "Not enough gems",
402            Self::TooManyRequests => "Too many requests",
403            Self::MessageTooLarge => "Message too large",
404            Self::InvalidRequest => "Invalid request",
405            Self::InternalError => "Internal server error",
406        }
407    }
408}
409
410impl std::fmt::Display for ErrorCode {
411    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
412        write!(f, "{}", self.message())
413    }
414}

439    #[test]
440    fn test_timer_vote_state_serialization() {
441        let idle = TimerVoteState::Idle;
442        let json = serde_json::to_string(&idle).unwrap();
443        assert!(json.contains(r#""status":"idle""#));
444
445        let now = chrono::Utc::now();
446        let vote = TimerVoteState::VoteInProgress {
447            initiator_id: "123".to_string(),
448            voters: vec!["456".to_string()],
449            votes_needed: 2,
450            expires_at: now,
451        };
452        let json = serde_json::to_string(&vote).unwrap();
453        assert!(json.contains(r#""status":"vote_in_progress""#));
454        assert!(json.contains(r#""initiator_id":"123""#));
455        assert!(json.contains(r#""expires_at""#));
456
457        let active = TimerVoteState::TimerActive {
458            expires_at: now,
459            target_player_id: "789".to_string(),
460        };
461        let json = serde_json::to_string(&active).unwrap();
462        assert!(json.contains(r#""status":"timer_active""#));
463        assert!(json.contains(r#""target_player_id":"789""#));
464
465        let cooldown = TimerVoteState::Cooldown { expires_at: now };
466        let json = serde_json::to_string(&cooldown).unwrap();
467        assert!(json.contains(r#""status":"cooldown""#));
468        assert!(json.contains(r#""expires_at""#));
469    }
